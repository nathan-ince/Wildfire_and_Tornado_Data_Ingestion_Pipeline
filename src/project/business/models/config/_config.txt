import logging

from dataclasses import dataclass
from typing import Self, cast

logger = logging.getLogger(__name__)

@dataclass
class Config:
  @dataclass
  class Internal:
    @dataclass
    class Tables:
      accepted: str
      rejected: str
      @classmethod
      def verify(cls, tables, entity_name: str = "tables") -> Self:
        """
        ### **Throws**

        `VerifyErrorName`
        """
        try:
          try:
            if not verify_python_dict_guard(tables, entity_name): raise
          except VerifyErrorName: raise
          try:
            accepted = verify_python_str_return(tables.get("accepted"), "accepted")
            rejected = verify_python_str_return(tables.get("rejected"), "rejected")
          except VerifyErrorName as e:
            error = VerifyErrorName(entity_name, e.message)
            logger.error(error.message)
            raise error from e
          return cls(accepted, rejected)
        except VerifyErrorName: raise
        except Exception as e:
          error = VerifyErrorName(entity_name, type(e).__name__)
          logger.error(error.message, extra={"entity_name": entity_name, "error_name": type(e).__name__})
          raise error from e
    @dataclass
    class Field:
      name: str
      ptype: PostgresDataType
      dtype: PyArrowDataType
      @classmethod
      def verify(cls, field, entity_name: str = "field") -> Self:
        try:
          try:
            if not verify_python_dict_guard(field, entity_name): raise
          except VerifyErrorName: raise
          try:
            name = verify_python_str_return(field.get("name"), "name")
            ptype = PostgresDataType.verify(field.get("ptype"), "ptype")
            dtype = PyArrowDataType.verify(field.get("dtype"), "dtype")
          except VerifyErrorName as e:
            error = VerifyErrorName(entity_name, e.message)
            logger.error(error.message)
            raise error from e
          return cls(name, ptype, dtype)
        except VerifyErrorName: raise
        except Exception as e:
          error = VerifyErrorName(entity_name, type(e).__name__)
          logger.error(error.message, extra={"entity_name": entity_name, "error_name": type(e).__name__})
          raise error from e
    tables: Tables
    fields: list[Field]
    @classmethod
    def verify(cls, internal, entity_name = "internal") -> Self:
      """
      ### **Throws**

      `VerifyErrorName`
      """
      try:
        try:
          if not verify_python_dict_guard(internal, entity_name): raise
        except VerifyErrorName: raise
        try:
          tables = cls.Tables.verify(internal.get("tables"), "tables")
          fields = verify_python_list_return(internal.get("fields"), "fields")
          for field_index in range(len(fields)):
            try: cls.Field.verify(fields[field_index])
            except VerifyErrorName as e:
              error = VerifyErrorName("fields", e.message)
              logger.error(error.message, extra={"entity_name": "fields", "field_index": field_index})
              raise error from e
          fields = list(cls.Field.verify(field) for field in verify_python_list_return(internal.get("fields"), "fields")) # necessary for now, though redundant
        except VerifyErrorName as e:
          error = VerifyErrorName(entity_name, e.message)
          logger.error(error.message)
          raise error from e
        return cls(tables, fields)
      except VerifyErrorName: raise
      except Exception as e:
        error = VerifyErrorName(entity_name, type(e).__name__)
        logger.error(error.message, extra={"entity_name": entity_name, "error_name": type(e).__name__})
        raise error from e
  @dataclass
  class External:
    @dataclass
    class Source:
      class Format(AbstractStringEnum):
        CSV = "csv"
        JSON = "json"
      @dataclass
      class Field:
        name: str
        dtype: PyArrowDataType
        @classmethod
        def verify(cls, field, entity_name: str = "field") -> Self:
          """
          ### **Throws**

          `VerifyErrorName`
          """
          try:
            try:
              if not verify_python_dict_guard(field, entity_name): raise
            except VerifyErrorName: raise
            try:
              name = verify_python_str_return(field.get("name"), "name")
              dtype = PyArrowDataType.verify(field.get("dtype"), "dtype")
            except VerifyErrorName as e:
              error = VerifyErrorName(entity_name, e.message)
              logger.error(error.message)
              raise error from e
            return cls(name, dtype)
          except VerifyErrorName: raise
          except Exception as e:
            error = VerifyErrorName(entity_name, type(e).__name__)
            logger.error(error.message, extra={"entity_name": entity_name, "error_name": type(e).__name__})
            raise error from e
      path: str
      format: str
      fields: list[Field]
      field_name_map: dict[str, str]
      @classmethod
      def verify(cls, source, entity_name: str = "source") -> Self:
        """
        ### **Throws**

        `VerifyErrorName`
        """
        try:
          try:
            if not verify_python_dict_guard(source, entity_name): raise
          except VerifyErrorName: raise
          try:
            path = verify_python_str_return(source.get("path"), "path")
            format = cls.Format.verify(source.get("format"), "format")
            fields = list(cls.Field.verify(field) for field in verify_python_list_return(source.get("fields"), "fields"))
            field_name_map = dict((verify_python_str_return(key, "field_name_map_key"), verify_python_str_return(value, "field_name_map_value")) for key, value in verify_python_dict_return(source.get("field_name_map"), "field_name_map").items())
          except VerifyErrorName as e:
            error = VerifyErrorName(entity_name, e.message)
            logger.error(error.message)
            raise error from e
          return cls(path, format, fields, field_name_map)
        except VerifyErrorName: raise
        except Exception as e:
          error = VerifyErrorName(entity_name, type(e).__name__)
          logger.error(error.message, extra={"entity_name": entity_name, "error_name": type(e).__name__})
          raise error from e
    sources: list[Source]
    @classmethod
    def verify(cls, external, entity_name: str = "external") -> Self:
      """
      ### **Throws**

      `VerifyErrorName`
      """
      try:
        try:
          if not verify_python_dict_guard(external, entity_name): raise
        except VerifyErrorName: raise
        try:
          sources = list(cls.Source.verify(source) for source in verify_python_list_return(external.get("sources"), "sources"))
        except VerifyErrorName as e:
          error = VerifyErrorName(entity_name, e.message)
          logger.error(error.message)
          raise error from e
        return cls(sources)
      except VerifyErrorName: raise
      except Exception as e:
        error = VerifyErrorName(entity_name, type(e).__name__)
        logger.error(error.message, extra={"entity_name": entity_name, "error_name": type(e).__name__})
        raise error from e
  internal: Internal
  external: External
  @classmethod
  def verify(cls, config, entity_name: str = "config") -> Self:
    """
    ### **Throws**

    `VerifyErrorName`
    """
    try:
      try:
        if not verify_python_dict_guard(config, entity_name): raise
      except VerifyErrorName: raise
      try:
        internal = cls.Internal.verify(config.get("internal"), "internal")
        external = cls.External.verify(config.get("external"), "external")
      except VerifyErrorName as e:
        error = VerifyErrorName(entity_name, e.message)
        logger.error(error.message)
        raise error from e
      source_count = len(external.sources)
      for source_index in range(source_count):
        try: verify_field_names(internal, external.sources[source_index])
        except VerifyErrorName as e:
          error = VerifyErrorName("source", e.message)
          logger.error(error.message, extra={"entity_name": "source", "source_index": source_index})
          raise error from e
      return cls(internal, external)
    except VerifyErrorName: raise
    except Exception as e:
      error = VerifyErrorName(entity_name, type(e).__name__)
      logger.error(error.message, extra={"entity_name": entity_name, "error_name": type(e).__name__})
      raise error from e

__all__ = ["Config"]

def verify_field_names(target: Config.Internal, source: Config.External.Source):
  target_field_name_set = set(field.name for field in target.fields)
  source_field_name_set = set(field.name for field in source.fields)
  source_field_name_map = source.field_name_map
  source_field_name_map_keys = source_field_name_map.keys()
  for source_field_name_map_key in source_field_name_map_keys:
    if source_field_name_map_key not in source_field_name_set:
      error = VerifyErrorName("source_field_name_map_key", "not found in source_field_name_set")
      logger.error(error.message, extra={"source_field_name_map_key": source_field_name_map_key, "source_field_name_set": source_field_name_set})
      raise error
    source_field_name_map_value = source_field_name_map.get(source_field_name_map_key)
    if source_field_name_map_value not in target_field_name_set:
      error = VerifyErrorName("source_field_name_map_value", "not found in target_field_name_set")
      logger.error(error.message, extra={"source_field_name_map_value": source_field_name_map_value, "target_field_name_set": target_field_name_set})
      raise error
